# Read in Harry Potter and the Sorcer's Stone
hp_text <- pdf_text("harry_potter_and_the_sorcerers_JKrowling.pdf")

Part 1: Text Analysis

# Make the text tidy
hp_tidy <- data.frame(hp_text) %>% 
  mutate(text_full = str_split(hp_text, pattern ="\\n")) %>% 
  unnest(text_full) %>% 
  mutate(text_full = str_trim(text_full))
# Make it even tidier!
hp_df <- hp_tidy %>% 
  slice(-(1:44)) %>% 
    mutate(chapter = case_when(
    str_detect(text_full, pattern = "CHAPTER") ~ text_full,
    TRUE ~ NA_character_
  )) %>% 
  fill(chapter) %>% 
  separate(col = chapter, into = c("ch", "no"), sep = " ") %>% 
  mutate(chapter = case_when(
    no == "ONE" ~ 1,
    no == "TWO" ~ 2,
    no == "THREE" ~ 3,
    no == "FOUR" ~ 4,
    no == "FIVE" ~ 5,
    no == "SIX" ~ 6,
    no == "SEVEN"~ 7,
    no == "EIGHT" ~ 8,
    no == "NINE" ~ 9,
    no == "TEN" ~ 10,
    no == "ELEVEN" ~ 11,
    no == "TWELVE" ~ 12,
    no == "THIRTEEN" ~ 13,
    no == "FOURTEEN" ~ 14,
    no == "FIFTEEN" ~ 15,
    no == "SIXTEEN" ~ 16,
    no == "SEVENTEEN" ~ 17
  )) %>% 
  mutate(chapter = as.numeric(chapter))
# Get tokens of the text
hp_tokens <- hp_df %>% 
  unnest_tokens(word, text_full) %>% 
  dplyr::select(-hp_text) 

# Get word counts
hp_wordcount <- hp_tokens %>% 
  count(chapter, word)
# Remove all stop words
hp_nonstop_words <- hp_tokens %>% 
  anti_join(stop_words)

# Get counts again
hp_nonstop_counts <- hp_nonstop_words %>% 
  count(chapter, word)
# Find the top 5 words from each chapter
hp_top5_words <- hp_nonstop_counts %>% 
  group_by(chapter) %>% 
  arrange(-n) %>% 
  slice(1:5)

ggplot(data = hp_top5_words, aes(x = word, y = n)) +
  geom_col(fill = "maroon") +
  facet_wrap(~chapter, scales = "free") +
  coord_flip() +
  labs(x = "Words", y = "Word Count", title = "Top 5 Words Used in Each Chapter", subtitle = "Text: Harry Potter and the Sorcerer's Stone")

Figure 1: The top 5 words used in each chapter of Harry Potter and the Sorcerers Stone by J.K. Rowling can reveal a lot about what occurs in each chapter. For instance, in chapter four the words hagrid and yeh are used a lot. This is because in that chapter Harry, the main character, meets Hagrid for the first time and Hagrid talks with an accent that often includes the phrase yeh.

Part 2: Sentiment Analyis

# Bring hp test and afinn lexicon together
hp_afinn <- hp_nonstop_words %>% 
  inner_join(get_sentiments("afinn"))

# Find counts based on snetiment rnaking from afinn lexicon
hp_afinn_counts <- hp_afinn %>% 
  count(chapter, value)

# Find the mean afinn score by chapter
hp_afinn_means <- hp_afinn %>% 
  group_by(chapter) %>% 
  summarize(mean_afinn = mean(value))
# Plot the hp afinn ranking (mean of each chapter)
ggplot(data = hp_afinn_means,
       aes(x = fct_rev(as.factor(chapter)),
           y = mean_afinn)) +
  geom_col(fill="maroon") +
  coord_flip() +
  labs(x = "Afinn Mean Score", y = "Chapter", title = "Mean Afinn Score of Each Chapter in Harry Potter and the Sorcerer's Stone")

Figure 3: Using the Afinn lexicon for a sentiment analysis reveals how negative or positive text can be. Words are rated from 1-3 (most negative) to 3 (most positive). By taking the average Afinn scores of each chapter, we can see how relatively negative or positive the text is. Harry Potter and the Sorcerers Stone is an adventurous book with a lot of ups and downs and fearful happenings so we see that on average the text appears to be relatively negative but not beyond -1 on average, meaning its not horribly scary, negative, sad, etc.
